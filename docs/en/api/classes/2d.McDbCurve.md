# Class: McDbCurve

[2d](../modules/2d.md).McDbCurve

Representing the database curve. Implemented operations related to curves, such as calculating the length, nearest point, area, tangential direction of any point on the curve, intersection points, coordinate transformation, interruption, offset, discretization, and other functions.

**`Example`**

```ts
//The following code demonstrates selecting a curve, then selecting another curve, calculating the intersection point between the curve and this curve, and then breaking the curve at the intersection point.
   import { MxCADUiPrEntity, MxCADResbuf, MxCpp, MxCADUtility, McDbCurve } from "mxcad";

   async function Mx_IntersectBreak() {
//Select reference curve
       let eventObj = new MxCADUiPrEntity();
EventObj.setMessage ("Please select reference curve");
       let eventObj_id = await eventObj.go();
Let event=await eventObj_id. getMcDbEntity ()//Get instance object
       if (event === null) return;
       const mxcad = MxCpp.App.getCurrentMxCAD();
let curve = event.clone();//  Clone Curve
       let filter = new MxCADResbuf();
//Set filters to select circles, arcs, lines, polylines, and ellipses
       filter.AddMcDbEntityTypes("CIRCLE,ARC,LINE,LWPOLYLINE,ELLIPSE");
Let aryId=await MxCADUtility. userSelect ("select interrupt object", filter);//ID of the selected object
       if (aryId.length == 0) return
//Traverse the selected curve object
       aryId.forEach(async (id) => {
           if (id.id !== eventObj_id.id) {
let breakEvent: McDbCurve = (await id.getMcDbEntity()) as McDbCurve;//  Obtain an instance of a broken curve
let breakArr = breakEvent.IntersectWith(curve as McDbEntity, McDb.Intersect.kOnBothOperands);// Set of points intersecting with entities
               if (breakArr.length() != 0) {
//Objects with intersection points
let arr: McGePoint3d[] = [];// Intersection array
                   breakArr.forEach((item: McGePoint3d) => {
                       arr.push(item)
                   })
                   let breakPoint: McGePoint3d[] = [];
                   arr.forEach(pt => {
let closePoint = breakEvent.getClosestPointTo(pt, false);// The point on the curve closest to the mouse position
                       if (!closePoint.ret) return;
let vec = breakEvent.getFirstDeriv(closePoint.val);// The vector at the location of the breakpoint
                       if (!vec.ret) return;
vec.val.normalize().mult(MxFun.viewCoordLong2Cad(10));// Distance of disconnection
                       let pt1 = closePoint.val.clone();
                       pt1.addvec(vec.val);
                       let pt2 = closePoint.val.clone();
                       pt2.subvec(vec.val);
                       breakPoint.push(pt1);
                       breakPoint.push(pt2);
                   });
let breakcurve = breakEvent.splitCurves(breakPoint);//  Cut the curve based on the endpoints
                   if (breakcurve.empty()) {
                       breakEvent.highlight(false);
                       return;
                   }
//Redraw the broken curve
                   breakcurve.forEach((obj: McDbObject, index: number) => {
                       if (index % 2 == 0) mxcad.drawEntity(obj as McDbEntity);
                   });
BreakEvent. erase()//Delete the original curve
               }
           }
       })
   } 
```

### explode

▸ **explode**(): [`MxCADResbuf`](2d.MxCADResbuf.md)

Break the object and return the linked list of the object's data after being broken

**`Example`**

```ts
import { McDbEntity, MxCADResbuf } from "mxcad";
//Obtain the target object
let getEnt = new MxCADUiPrEntity();
GetEnt.setMessage ("Select object to break: ");
let id = await getEnt.go();
let ent:McDbEntity = id.getMcDbEntity();
if (ent === null) return;
//Smash objects
      let retExplode: MxCADResbuf = ent.explode();
      if (retExplode.GetCount() == 0) return;
      let iExplodeConut = retExplode.GetCount();
      for (let j = 0; j < iExplodeConut; j++) {
          let tmpobj = retExplode.AtObject(j).val;
          if(tmpobj instanceof McDbEntity ){
              mxcad.drawEntity(tmpobj);
          }
      }
```

#### Returns

[`MxCADResbuf`](2d.MxCADResbuf.md)
